找出字符串中连续出现最多的字符和个数
```
function getMaxLength(str){
    let c = ''
    let max = 0
    let res = {}
    let map = {}
    let ret = Array.from(str).reduce((acc,cur) => {
        if(c === cur){
            acc[cur] += 1
        }else{
            acc[cur] = 1
        }
        c = cur
        return acc
    },{})
    max = Math.max.apply(null, Object.values(ret))
    for(let i in ret){
        if(ret[i] === max){
            res[i] = max
        }
    }
    return  res
}
```
