### 判断两个对象是否相等
https://juejin.im/post/598a701b6fb9a03c5b04bb14#heading-5
```
function eq(obj1, obj2){
    const keys1 = Object.keys(obj1)
    const keys2 = Object.keys(obj2)
    if(keys1.length !== keys2.length){
        return false
    } 
    let ret
    keys1.map(key1 => {
        const propName = key1
        const prop1 = obj1[propName]
        const prop2 = obj2[propName]
        if(typeof prop1 === 'object' && prop1 !== null){
            if(eq(prop1, prop2)){
                ret = true
            }else {
                ret = false
            }
        }else if(prop1 !== prop2){
            ret = false
        }else {
            ret = true
        }
    })
    return ret
}
```
