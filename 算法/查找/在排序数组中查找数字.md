### 排序数组中查找数字
统计一个数字在排序数组中出现的次数   
找到第一次和最后一次的位置
```
function getNum(data, trget){
    if(data && data.length > 0 && target){
        const fistIndex = getFirst(data, 0, data.length - 1, trget)
        const lastIndex = getLat(data, 0, data.length - 1, trget)
        if(firstIndex !== -1 && lastIndex !== -1){
            return lastIndex - firstIndex + 1
        }
    }
    return 0
}
function getFirst(data, first, last, target){
    if(first > last){
        return -1
    }
    const mid = Math.floor((first + last) / 2)
    if(data[mid] === target){
        if(data[mid - 1] !== target){
            return mid
        }else {
            return getFirst(data, first, mid - 1, target)
        }
    }else if(data[mid] > target){
        return getFirst(data, first, mid - 1, target)
    }else if(data[mid] < target){
        return getFirst(data, mid + 1, last, target)
    }
}
function getLast(data, first, last, target){
    if(first > last){
        return -1
    }
    const mid = Math.floor((first + last) / 2)
    if(data[mid] === target){
        if(data[mid + 1] !== target){
            return mid
        }else {
            return getLast(data, mid + 1, last, target)
        }
    }else if(data[mid] > target){
        return getFirst(data, first, mid - 1, target)
    }else if(data[mid] < target){
        return getFirst(data, mid + 1, last, target)
    }
}
```
